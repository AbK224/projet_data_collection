pages = st.number_input("Nombre de pages à scraper", min_value=1, max_value=100, value=1)

if st.button("Scrapper les données"):
    for i in range(1,5):
        if i == 1:
            df = scrap_dog_data(pages)
            st.write(f'Pour les chiens, on a {df.shape[0]} lignes et {df.shape[1]} colonnes')
            st.write(df)
        elif i == 2:
            df2 = scrap_dog_sheeps(pages)
            st.write(f'Pour les moutons, on a {df2.shape[0]} lignes et {df2.shape[1]} colonnes')
            st.write(df2)
        elif i == 3:
            df3 = scrap_dog_animals(pages)
            st.write(df3)
            st.write(f'Pour les poules, lapins et pigeons, on a {df3.shape[0]} lignes et {df3.shape[1]} colonnes')
        elif i == 4:
            df4 = scrap_dog_others(pages)
            st.write(df4)
            st.write(f'Pour les autres animaux, on a {df4.shape[0]} lignes et {df4.shape[1]} colonnes')
